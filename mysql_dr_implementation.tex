\section{Overview}
This document outlines the step-by-step implementation for the \textbf{Double-Instance} Disaster Recovery (DR) architecture. In this design, both \textbf{Click} and \textbf{Pleniged} databases are replicated to both Zone 2 and Zone 3 servers using isolated Docker containers on different ports.

\section{Infrastructure Layout}
\begin{itemize}
    \item \textbf{Zone 2 (Primary DR)}:
        \begin{itemize}
            \item \texttt{mysql-click} container on Port 3306 (Semi-Sync).
            \item \texttt{mysql-pleniged} container on Port 3307 (Semi-Sync).
        \end{itemize}
    \item \textbf{Zone 3 (Standby DR)}:
        \begin{itemize}
            \item \texttt{mysql-click} container on Port 3306 (Async).
            \item \texttt{mysql-pleniged} container on Port 3307 (Async).
        \end{itemize}
\end{itemize}

\section{Jira Epics and Tasks}

\subsection{Epic 1: DR Infrastructure Preparation (Multi-Container)}
\begin{itemize}[label=$\square$]
    \item \textbf{Task 1.1}: Provision Ubuntu 22.04 servers for Zone 2 and Zone 3.
    \item \textbf{Task 1.2}: Install Docker and Docker-Compose on both DR zones.
    \item \textbf{Task 1.3}: Configure firewall rules for Ports 3306 and 3307 across all zones.
\end{itemize}

\subsection{Epic 2: Click Replication Setup}
\begin{itemize}[label=$\square$]
    \item \textbf{Task 2.1}: Deploy \texttt{mysql-click} container on Zone 2 and Zone 3.
    \item \textbf{Task 2.2}: Enable Semi-Sync replication from Source to \textbf{Zone 2 :3306}.
    \item \textbf{Task 2.3}: Enable Async replication from Source to \textbf{Zone 3 :3306}.
\end{itemize}

\subsection{Epic 3: Pleniged Replication Setup}
\begin{itemize}[label=$\square$]
    \item \textbf{Task 3.1}: Deploy \texttt{mysql-pleniged} container on Zone 2 and Zone 3 (Port 3307).
    \item \textbf{Task 3.2}: Enable Semi-Sync replication from Source to \textbf{Zone 2 :3307}.
    \item \textbf{Task 3.3}: Enable Async replication from Source to \textbf{Zone 3 :3307}.
\end{itemize}

\subsection{Epic 4: Monitoring \& Observability}
\begin{itemize}[label=$\square$]
    \item \textbf{Task 4.1}: Deploy \texttt{mysqld\_exporter} for Port 3306 (Listen on 9104).
    \item \textbf{Task 4.2}: Deploy \texttt{mysqld\_exporter} for Port 3307 (Listen on 9105).
    \item \textbf{Task 4.3}: Update Prometheus config on Zone 1 to scrape all 4 targets.
\end{itemize}
